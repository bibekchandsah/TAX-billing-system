rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    match /users/{userId} {
      // Helper function to check if user is admin
      function isAdmin() {
        return request.auth != null && request.auth.token.email == 'bibekchandsah@gmail.com';
      }

      // User can read/write their own profile document (stores photoData etc.), admin can read
      allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
      allow write: if request.auth != null && request.auth.uid == userId;

      // =========================================================================
      // RULES FOR INVOICE BILLING SYSTEM
      // =========================================================================
      
      // User can read/write their own bills subcollection, admin can read
      match /bills/{billId} {
        allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
        allow write: if request.auth != null && request.auth.uid == userId;
      }

      // User can read/write their own settings subcollection, admin can read
      match /settings/{settingId} {
        allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
        allow write: if request.auth != null && request.auth.uid == userId;
      }

      // User can read/write their own sessions subcollection, admin can read
      match /sessions/{sessionId} {
        allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
        allow write: if request.auth != null && request.auth.uid == userId;
      }

      // Customers and ledger
      match /customers/{customerId} {
        allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
        allow write: if request.auth != null && request.auth.uid == userId;

        match /ledger/{ledgerId} {
          allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
          allow write: if request.auth != null && request.auth.uid == userId;
        }
      }

      // Parties and ledger
      match /parties/{partyId} {
        allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
        allow write: if request.auth != null && request.auth.uid == userId;

        match /ledger/{ledgerId} {
          allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
          allow write: if request.auth != null && request.auth.uid == userId;
        }
      }

      // Suppliers and ledger
      match /suppliers/{supplierId} {
        allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
        allow write: if request.auth != null && request.auth.uid == userId;

        match /ledger/{ledgerId} {
          allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
          allow write: if request.auth != null && request.auth.uid == userId;
        }
      }

      // Stock and ledger
      match /stock/{particularId} {
        allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
        allow write: if request.auth != null && request.auth.uid == userId;
        
        match /ledger/{ledgerId} {
          allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
          allow write: if request.auth != null && request.auth.uid == userId;
        }
      }

      // =========================================================================
      // RULES FOR VAT BILLING SYSTEM (ISOLATED WORKSPACE)
      // =========================================================================
      
      // Allow access to the isolated VAT_PAN workspace
      match /VAT_PAN/data {
        match /records/{recordId} {
          allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
          allow write: if request.auth != null && request.auth.uid == userId;
        }
        match /ledger/{ledgerId} {
          allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
          allow write: if request.auth != null && request.auth.uid == userId;
        }
        match /stock/{stockId} {
          allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
          allow write: if request.auth != null && request.auth.uid == userId;
        }
        match /settings/{settingId} {
          allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
          allow write: if request.auth != null && request.auth.uid == userId;
        }
        match /sessions/{sessionId} {
          allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
          allow write: if request.auth != null && request.auth.uid == userId;
        }
      }
    }
  }
}
